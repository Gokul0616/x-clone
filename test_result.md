```javascript
// Multi-factor recommendation scoring
const RecommendationEngine = {
  // Content-based filtering
  contentScore: user.favoriteHashtags ‚à© tweet.hashtags * 2 +
                user.communities ‚à© tweet.communityId * 3 +
                user.interactedUsers ‚à© tweet.userId * 2,
  
  // Collaborative filtering  
  collaborativeScore: similarUsers ‚à© tweet.likedBy * 1.5 +
                     similarUsers ‚à© tweet.retweetedBy * 2,
  
  // Time decay (freshness)
  timeDecay: Math.exp(-hoursOld / 24),
  
  // Quality scoring
  qualityScore: engagementRate * 10,
  
  // Final score with weights
  finalScore: (contentScore * 0.4 + collaborativeScore * 0.2 + 
              engagementScore * 0.3) * timeDecay + qualityScore * 0.1
}
```
